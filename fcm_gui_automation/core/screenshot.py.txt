from pathlib import Path
from datetime import datetime


def build_screenshot_path(base_dir: Path, name: str) -> Path:
    report_root = base_dir.parent if base_dir.name == "fcm_gui_automation" else base_dir
    screenshots_dir = report_root / "reports" / "screenshots"
    screenshots_dir.mkdir(parents=True, exist_ok=True)
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    return screenshots_dir / f"{timestamp}_{name}.png"
